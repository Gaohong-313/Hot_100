class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        vector<vector<int>> result;
        vector<int> path;
        // 启动回溯
        versal(nums, path, result, 0);
        return result;
    }

    // 注意：star 参数代表当前可以从 nums 的哪个下标开始选择
    void versal(vector<int>& nums, vector<int>& path, vector<vector<int>>& result, int star) {
        // 【修正点 1】：每次进入递归，首先记录当前的路径
        // 这样第一次调用时 path 是空的，就会把空集加入结果
        result.push_back(path);

        // 从 star 开始遍历，确保不回头选前面的元素，避免重复
        for (int i = star; i < nums.size(); ++i) {
            // 1. 做选择
            path.push_back(nums[i]);
            
            // 2. 进入下一层
            // 【修正点 2】：这里必须传 i + 1
            // 表示下一层只能从当前元素的后面开始选，不能重复选，也不能回头选
            versal(nums, path, result, i + 1);
            
            // 3. 撤销选择 (回溯)
            path.pop_back();
        }
    }
};
